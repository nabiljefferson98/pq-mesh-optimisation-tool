% Figure 4.2 — Log-log scalability plot
% ---------------------------------------------------------------------------
\begin{figure}[htbp]
  \centering
  \includegraphics[width=0.85\textwidth]{EXP-01_scalability_loglog}
  \caption{Log-log plot of wall-clock optimisation time versus number of
    vertices for synthetic planar grid meshes (Series~A, Machine~A).
    The fitted complexity line gives $T(n) \approx 0.0007 \cdot n^{1.27}$
    ($R^2 = 1.000$), confirming sub-quadratic empirical complexity.}
  \label{fig:scalability_loglog}
\end{figure}

% ---------------------------------------------------------------------------
% Figure 4.3 — Convergence and scaling overlay
% ---------------------------------------------------------------------------
\begin{figure}[htbp]
  \centering
  \includegraphics[width=0.85\textwidth]{EXP-01_convergence_scaling}
  \caption{Energy convergence curves and scaling overlay for Series~A
    (synthetic grids) and Series~B (oloid resolution series) on the same
    log-log axes. The consistent slope across both series demonstrates that
    the empirical complexity exponent $n^{1.27}$ generalises beyond
    structured synthetic grids to a real-world developable surface.}
  \label{fig:convergence_scaling}
\end{figure}

% ---------------------------------------------------------------------------
% Figure 4.8 — EXP-05 planarity deviation histograms
% ---------------------------------------------------------------------------
\begin{figure}[htbp]
  \centering
  \includegraphics[width=\textwidth]{EXP-05_planarity_histograms}
  \caption{Per-face planarity deviation $|d_f|$ histograms before (grey)
    and after (blue) optimisation for all four EXP-05 real-world meshes
    (Spot, Blub, Oloid-256, Bob). The post-optimisation distributions are
    tightly concentrated near zero for all meshes, confirming successful
    convergence. The oloid histogram is truncated at $|d_f| = 0.01$ because
    the surface is already near-developable prior to optimisation.}
  \label{fig:planarity_histograms}
\end{figure}

% ---------------------------------------------------------------------------
% Figure 4.9 — Oloid spatial heatmap
% ---------------------------------------------------------------------------
\begin{figure}[htbp]
  \centering
  \includegraphics[width=0.80\textwidth]{EXP-05_oloid_spatial_heatmap}
  \caption{Spatial distribution of per-face planarity deviation $|d_f|$
    on the optimised oloid-256 mesh rendered as a colour heatmap
    (blue~$= 0$, red~$= \max$). Residual deviation is concentrated along
    the two circular boundary curves, consistent with the theoretical
    prediction that the ruling regions of a developable surface must
    achieve zero Gaussian curvature under planar-quad constraints.}
  \label{fig:oloid_heatmap}
\end{figure}
